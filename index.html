let perguntaAtual = 0; // Função para mostrar a pergunta e alternativas function mostrarPergunta() { // Verifica se há mais perguntas if (perguntaAtual < perguntas.length) { const pergunta = perguntas[perguntaAtual]; // Exibe o enunciado document.getElementById('pergunta').textContent = pergunta.enunciado; // Cria botões para as alternativas caixaAlternativas.innerHTML = ''; // Limpa as alternativas anteriores pergunta.alternativas.forEach((alternativa) => { const button = document.createElement('button'); button.textContent = alternativa; button.onclick = () => mostrarResultado(alternativa); caixaAlternativas.appendChild(button); }); } else { // Exibe o resultado final se não houver mais perguntas document.getElementById('pergunta').textContent = 'Obrigado por responder!'; // Avança para a próxima pergunta perguntaAtual++; setTimeout(mostrarPergunta, 1000); // Aguarda 1 segundo antes de mostrar a próxima pergunta } // Inicia com a primeira pergunta mostrarPergunta();
